#include <ESP8266WiFi.h>
#include <espnow.h>
#include <Wire.h>
#include <Adafruit_BMP085.h>

#define MPU_ADDR 0x68
#define MQ135_PIN A0

Adafruit_BMP085 bmp;

// ⚠️ CRITICAL: Replace with YOUR RECEIVER'S MAC Address!
// Get this from Receiver NodeMCU's Serial Monitor
uint8_t receiverMAC[] = {0xEC, 0x64, 0xC9, 0xCD, 0xEE, 0x7C};

typedef struct {
  int16_t ax, ay, az;
  int16_t gx, gy, gz;
  float pressure;
  float bmpTemp;
  int mq135;
} Telemetry;

Telemetry data;

void onDataSent(uint8_t *mac, uint8_t status) {
  if (status == 0) {
    Serial.println("✅ ESP-NOW: SUCCESS");
  } else {
    Serial.print("❌ ESP-NOW: FAIL (Code ");
    Serial.print(status);
    Serial.println(")");
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n=== SATELLITE TRANSMITTER ===");
  
  // I2C setup
  Wire.begin(4, 5);  // D2=SDA, D1=SCL
  Wire.setClock(400000);
  
  // MPU6050 setup
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x6B);
  Wire.write(0);
  Wire.endTransmission(true);
  Serial.println("✅ MPU6050 ready");
  
  // BMP180 setup
  if (!bmp.begin()) {
    Serial.println("❌ BMP180 NOT FOUND!");
    while(1);
  }
  Serial.println("✅ BMP180 ready");
  
  // ESP-NOW setup
  WiFi.mode(WIFI_STA);
  
  if (esp_now_init() != 0) {
    Serial.println("❌ ESP-NOW init failed");
    while(1);
  }
  
  esp_now_set_self_role(ESP_NOW_ROLE_CONTROLLER);
  esp_now_register_send_cb(onDataSent);
  
  // ⚠️ FIXED: Added WiFi channel parameter
  esp_now_add_peer(receiverMAC, ESP_NOW_ROLE_SLAVE, 6, NULL, 0);
  
  Serial.print("📡 Transmitter MAC: ");
  Serial.println(WiFi.macAddress());
  Serial.println("🚀 Ready to transmit!");
}

void loop() {
  // Read MPU6050
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x3B);
  Wire.endTransmission(false);
  Wire.requestFrom((uint8_t)MPU_ADDR, (size_t)14);
  
  data.ax = Wire.read() << 8 | Wire.read();
  data.ay = Wire.read() << 8 | Wire.read();
  data.az = Wire.read() << 8 | Wire.read();
  Wire.read(); Wire.read(); // Skip temperature
  data.gx = Wire.read() << 8 | Wire.read();
  data.gy = Wire.read() << 8 | Wire.read();
  data.gz = Wire.read() << 8 | Wire.read();
  
  // Read BMP180
  data.pressure = bmp.readPressure() / 100.0;
  data.bmpTemp = bmp.readTemperature();
  
  // Read MQ135
  data.mq135 = analogRead(MQ135_PIN);
  
  // Send via ESP-NOW
  esp_now_send(receiverMAC, (uint8_t*)&data, sizeof(data));
  
  // Print to Serial Monitor
  Serial.println("\n----- TELEMETRY -----");
  Serial.print("AX: "); Serial.print(data.ax);
  Serial.print(" AY: "); Serial.print(data.ay);
  Serial.print(" AZ: "); Serial.println(data.az);
  Serial.print("Temp: "); Serial.print(data.bmpTemp, 1); Serial.println(" °C");
  Serial.print("MQ135: "); Serial.println(data.mq135);
  Serial.println("---------------------");
  
  delay(1000);
}
