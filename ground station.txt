// ESP-NOW callback
void onDataRecv(uint8_t *mac, uint8_t *incomingData, uint8_t len) {
  if (len == sizeof(Telemetry)) {
    memcpy(&rx, incomingData, sizeof(rx));
    lastDataTime = millis();
    
    // Print to Serial Monitor
    Serial.print("📥 Received: AX=");
    Serial.print(rx.ax);
    Serial.print(", Temp=");
    Serial.print(rx.bmpTemp, 1);
    Serial.print("°C, MQ=");
    Serial.println(rx.mq135);
  }
}

// HTTP handlers
void handleRoot() {
  server.send_P(200, "text/html", dashboardHTML);
}

void handleData() {
  String json = "{";
  json += "\"ax\":" + String(rx.ax) + ",";
  json += "\"ay\":" + String(rx.ay) + ",";
  json += "\"az\":" + String(rx.az) + ",";
  json += "\"gx\":" + String(rx.gx) + ",";
  json += "\"gy\":" + String(rx.gy) + ",";
  json += "\"gz\":" + String(rx.gz) + ",";
  json += "\"bmpTemp\":" + String(rx.bmpTemp) + ",";
  json += "\"pressure\":" + String(rx.pressure) + ",";
  json += "\"mq135\":" + String(rx.mq135) + ",";
  json += "\"timestamp\":" + String(millis());
  json += "}";
  
  // Add CORS headers
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.send(200, "application/json", json);
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n=== SATELLITE RECEIVER ===");
  
  // Setup WiFi hotspot
  WiFi.mode(WIFI_AP_STA);
  WiFi.softAP(ssid, password);
  
  // Configure AP IP
  IPAddress apIP(192, 168, 4, 1);
  IPAddress gateway(192, 168, 4, 1);
  IPAddress subnet(255, 255, 255, 0);
  WiFi.softAPConfig(apIP, gateway, subnet);
  
  Serial.println("📡 WiFi Hotspot Created:");
  Serial.print("   SSID: "); Serial.println(ssid);
  Serial.print("   Password: "); Serial.println(password);
  Serial.print("   IP Address: "); Serial.println(WiFi.softAPIP());
  
  // Print MAC address
  Serial.print("📱 Receiver MAC: ");
  Serial.println(WiFi.macAddress());
  
  // Setup ESP-NOW
  if (esp_now_init() != 0) {
    Serial.println("❌ ESP-NOW init failed!");
    delay(1000);
    ESP.restart();
  }
  
  // Set WiFi channel to match transmitter
  wifi_set_channel(6);
  
  esp_now_set_self_role(ESP_NOW_ROLE_SLAVE);
  esp_now_register_recv_cb(onDataRecv);
  
  Serial.println("✅ ESP-NOW ready (waiting for data...)");
  
  // Setup web server
  server.on("/", handleRoot);
  server.on("/data", handleData);
  
  server.begin();
  Serial.println("✅ Web server started");
  
  Serial.println("\n🔗 HOW TO CONNECT:");
  Serial.println("   1. Connect laptop to WiFi: SATELLITE_DASH");
  Serial.println("   2. Password: 12345678");
  Serial.println("   3. Open browser to: http://192.168.4.1");
  Serial.println("========================================\n");
}

void loop() {
  server.handleClient();
  
  // Print status every 5 seconds
  static unsigned long lastPrint = 0;
  if (millis() - lastPrint > 5000) {
    lastPrint = millis();
    
    if (millis() - lastDataTime > 5000) {
      Serial.println("⚠️  Waiting for transmitter data...");
    } else {
      Serial.print("✅ Receiving data (");
      Serial.print((millis() - lastDataTime) / 1000);
      Serial.println(" seconds ago)");
    }
  }
  
  delay(10);
}
