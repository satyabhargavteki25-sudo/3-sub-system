void crypt(char *data) {
  for (int i = 0; i < strlen(data); i++) data[i] ^= ENC_KEY;

void onReceive(const esp_now_recv_info *info, const uint8_t *data, int len) {
  Packet p; 
  memcpy(&p, data, sizeof(p));
  
  if(p.type == 0) { // Message from G2
    crypt(p.msg);
    
    // Get current time
    char timestamp[20];
    unsigned long millisTime = millis();
    int hours = (millisTime / 3600000) % 24;
    int minutes = (millisTime / 60000) % 60;
    int seconds = (millisTime / 1000) % 60;
    sprintf(timestamp, "%02d:%02d:%02d", hours, minutes, seconds);
    
    // Add to history with timestamp
    String msgHTML = "<div class='message g2'>";
    msgHTML += "<div class='message-header'>";
    msgHTML += "<span class='sender'>G2</span>";
    msgHTML += "<span class='timestamp'>" + String(timestamp) + "</span>";
    msgHTML += "</div>";
    msgHTML += "<div class='message-content'>" + String(p.msg) + "</div>";
    msgHTML += "</div>";
    
    history = msgHTML + history; // Add to beginning
    
    // Send acknowledgment back
    Packet ack = {1, 2, 1, "DELIVERED"};
    esp_now_send(SAT_MAC, (uint8_t*)&ack, sizeof(ack));
    
  } else if (p.type == 1) { // Delivery acknowledgment
    // Update last message with delivery status
    String deliveredHTML = "<div class='message system'>";
    deliveredHTML += "✓ Message delivered to G2";
    deliveredHTML += "</div>";
    
    history = deliveredHTML + history;
  }
}

void setup() {
  Serial.begin(115200);
  
  // Set up WiFi
  WiFi.mode(WIFI_AP_STA);
  WiFi.softAP("G1_NET", "12345678");
  esp_wifi_set_channel(1, WIFI_SECOND_CHAN_NONE);
  
  // Initialize ESP-NOW
  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }
  
  esp_now_register_recv_cb(onReceive);
  
  // Register peer
  esp_now_peer_info_t peer = {};
  memcpy(peer.peer_addr, SAT_MAC, 6);
  peer.channel = 1;
  peer.encrypt = false;
  
  if (esp_now_add_peer(&peer) != ESP_OK) {
    Serial.println("Failed to add peer");
    return;
  }
  
  // Set up web server routes
  server.on("/", []() {
    server.send_P(200, "text/html", page);
  });
  
  server.on("/send", []() {
    if (server.hasArg("m")) {
      packet = {1, 2, 0, ""};
      server.arg("m").toCharArray(packet.msg, 200);
      
      // Add timestamp for outgoing message
      unsigned long millisTime = millis();
      int hours = (millisTime / 3600000) % 24;
      int minutes = (millisTime / 60000) % 60;
      int seconds = (millisTime / 1000) % 60;
      char timestamp[20];
      sprintf(timestamp, "%02d:%02d:%02d", hours, minutes, seconds);
      
      String msgHTML = "<div class='message you'>";
      msgHTML += "<div class='message-header'>";
      msgHTML += "<span class='sender'>You (G1)</span>";
      msgHTML += "<span class='timestamp'>" + String(timestamp) + "</span>";
      msgHTML += "</div>";
      msgHTML += "<div class='message-content'>" + server.arg("m") + "</div>";
      msgHTML += "</div>";
      
      history = msgHTML + history;
      
      // Encrypt and send
      crypt(packet.msg);
      esp_now_send(SAT_MAC, (uint8_t*)&packet, sizeof(packet));
    }
    server.send(200, "text/plain", "OK");
  });
  
  server.on("/hist", []() {
    server.send(200, "text/html", history);
  });
  
  server.on("/clear", []() {
    history = "";
    server.send(200, "text/plain", "Cleared");
  });
  
  server.begin();
  
  Serial.println("G1 ESP-NOW Messenger started");
  Serial.print("AP SSID: ");
  Serial.println("G1_NET");
  Serial.print("AP IP address: ");
  Serial.println(WiFi.softAPIP());
  Serial.print("Target MAC (G2): ");
  for (int i = 0; i < 6; i++) {
    Serial.print(SAT_MAC[i], HEX);
    if (i < 5) Serial.print(":");
  }
  Serial.println();
  Serial.println("Ready to communicate with G2...");
}

void loop() { 
  server.handleClient();
